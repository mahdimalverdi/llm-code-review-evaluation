  \draw[draw=gridLine, line width=0.35pt, fill=white] (\x,\headerh) rectangle ++(\cellw,-\headerh);%

\newcommand{\rowscell}[2]{%
  \pgfmathsetmacro{\x}{\xstart + (#1) * \cellw}%
  \pgfmathsetmacro{\y}{\ystart - (#2) * \cellh}%
  \draw[draw=gridLine, line width=0.35pt, fill=rowShade] (\x,\y) rectangle ++(\cellw,-\cellh);%
}

\newcommand{\paperscell}[2]{%
  \pgfmathsetmacro{\x}{\xstart + (#1) * \cellw}%
  \pgfmathsetmacro{\y}{\ystart - (#2) * \cellh}%

\newcommand{\colheader}[2]{%
  \pgfmathsetmacro{\x}{\xstart + (#1 + 0.5) * \cellw}%
  \pgfmathsetmacro{\y}{0.5 * \headerh}%

}

\newcommand{\primaryfocus}[2]{%
  \pgfmathsetmacro{\x}{\xstart + (#1 + 0.5) * \cellw}%
  \pgfmathsetmacro{\y}{\ystart - (#2 + 0.5) * \cellh}%
  \fill[primaryFocus] (\x,\y) circle (0.105);%
}
\newcommand{\partialfocus}[2]{%
  \pgfmathsetmacro{\x}{\xstart + (#1 + 0.5) * \cellw}%
  \pgfmathsetmacro{\y}{\ystart - (#2 + 0.5) * \cellh}%
  \draw[partialFocus, line width=0.85pt, fill=partialFocus!18] (\x,\y) circle (0.105);%
  \draw[partialFocus, line width=0.85pt] (\x-0.075,\y) -- (\x+0.075,\y);%
}
\newcommand{\limitedfocus}[2]{%
  \pgfmathsetmacro{\x}{\xstart + (#1 + 0.5) * \cellw}%
  \pgfmathsetmacro{\y}{\ystart - (#2 + 0.5) * \cellh}%
  \draw[limitedFocus, line width=0.85pt] (\x,\y) circle (0.105);%
}
